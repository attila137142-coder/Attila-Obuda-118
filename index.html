LISTA + KERESŐ // ----------------------------- function renderList(data) { const list = document.getElementById('list'); list.innerHTML = ''; data.slice(0, 200).forEach(s => { const div = document.createElement('div'); div.className = 'stop'; div.innerText = s.name; div.onclick = () => { map.setView([s.lat, s.lon], 16); L.popup() .setLatLng([s.lat, s.lon]) .setContent(`${s.name}`) .openOn(map); }; list.appendChild(div); }); } // ----------------------------- // 4. TÉRKÉP MARKEREK // ----------------------------- function renderMarkers(data) { markers.forEach(m => map.removeLayer(m)); markers = []; data.forEach(s => { if (!s.lat || !s.lon) return; const marker = L.marker([s.lat, s.lon]) .addTo(map) .bindPopup(`${s.name}`); markers.push(marker); }); } // ----------------------------- // 5. KERESÉS // ----------------------------- const search = document.getElementById('search'); search.addEventListener('input', e => { const q = e.target.value.toLowerCase(); const filtered = stops.filter(s => s.name.toLowerCase().includes(q) ); renderList(filtered); renderMarkers(filtered); }); // ----------------------------- // START // ----------------------------- loadStops();
