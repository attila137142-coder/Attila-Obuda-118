// ====================== // IDŐ SZÁMÍTÁS // ====================== function getMinutes(t){ const now = new Date(); let [h,m] = (t || "0:0").split(":").map(Number); // GTFS miatt (25:30 stb) h = h % 24; const target = h*60 + m; const current = now.getHours()*60 + now.getMinutes(); return target - current; } // ====================== // MENETREND MEGJELENÍTÉS // ====================== function renderTimes(next){ let html = ""; next.forEach(n=>{ const min = getMinutes(n.time); if(min < 0) return; let cls = ""; if(min < 2) cls = "red"; else if(min > 10) cls = "blue"; html += `
🚍 ${n.line} | ${min} perc múlva
`; }); return html; } // ====================== // AUTO FRISSÍTÉS // ====================== setInterval(()=>{ render(); }, 30000);
