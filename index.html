const GTFS_URL = "https://bkk.hu/gtfs/budapest_gtfs.zip"; let stopsMap = {}; let tripsMap = {}; let routesMap = {}; let stopTimes = []; // ===== BETÖLTÉS async function loadGTFS(){ console.log("GTFS töltés..."); const res = await fetch(GTFS_URL); const blob = await res.blob(); const zip = await JSZip.loadAsync(blob); // ===== STOPS let stopsTxt = await zip.file("stops.txt").async("string"); stopsTxt.split("\n").slice(1).forEach(r=>{ let c=r.split(","); stopsMap[c[0]] = { name:c[2], lat:parseFloat(c[4]), lon:parseFloat(c[5]) }; }); // ===== ROUTES let routesTxt = await zip.file("routes.txt").async("string"); routesTxt.split("\n").slice(1).forEach(r=>{ let c=r.split(","); routesMap[c[0]] = c[2]; // route_short_name }); // ===== TRIPS let tripsTxt = await zip.file("trips.txt").async("string"); tripsTxt.split("\n").slice(1).forEach(r=>{ let c=r.split(","); tripsMap[c[2]] = { route:c[0], headsign:c[3] }; }); // ===== STOP TIMES let stTxt = await zip.file("stop_times.txt").async("string"); stopTimes = stTxt.split("\n").slice(1).map(r=>r.split(",")); console.log("GTFS kész"); renderReal(); } // ===== SZŰRÉS A TE 45 MEGÁLLÓDRA const myStops = [ "H5-09183","BKK-F00457","BKK-00438","BKK-F00611" ]; // ===== RENDER function renderReal(){ let now = new Date(); let timeNow = now.toTimeString().slice(0,8); let html=""; stopTimes.forEach(st=>{ let tripId = st[0]; let dep = st[2]; let stopId = st[3]; if(!myStops.includes(stopId)) return; if(dep < timeNow) return; let trip = tripsMap[tripId]; if(!trip) return; let route = routesMap[trip.route] || "?"; html += `
🚌 ${route} | ${dep} | ${trip.headsign}
`; }); document.body.innerHTML += html; } // ===== INDÍTÁS loadGTFS();
