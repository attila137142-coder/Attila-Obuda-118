const file = zipData.files["stops.txt"]; if(!file){ log("❌ stops.txt nincs!"); return; } const text = await file.async("string"); stops = parseCSV(text); log("📍 Stops: " + stops.length); drawStops(); }catch(e){ log("❌ HIBA: " + e.message); } } // DRAW STOPS function drawStops(){ log("🗺️ Rajzolás..."); markersLayer.clearLayers(); stops.slice(0, 800).forEach(s=>{ const lat = parseFloat(s.stop_lat); const lon = parseFloat(s.stop_lon); if(!lat || !lon) return; const m = L.circleMarker([lat,lon],{radius:4}) .bindPopup(`${s.stop_name}
`) .on("click", ()=>showStop(s.stop_id)); markersLayer.addLayer(m); }); log("✅ Megállók kész (800)"); } // STOP CLICK = "GÁZPEDÁL" function showStop(id){ const stop = stops.find(s=>s.stop_id===id); if(!stop) return; map.setView([stop.stop_lat, stop.stop_lon], 15); log("🚏 Stop: " + stop.stop_name); log("🚇 (itt jönne FUTÁR indulás lista - következő lépés)"); } // SEARCH function searchStops(){ const q = document.getElementById("search").value.toLowerCase(); const res = stops.filter(s=> (s.stop_name||"").toLowerCase().includes(q) ).slice(0,10); log("🔍 Találatok: " + res.length); res.forEach(r=>log("➡ " + r.stop_name)); if(res[0]){ map.setView([res[0].stop_lat, res[0].stop_lon], 15); } } // INIT initMap();
